import random

members = ['John', 'Mary', 'Mosh']

for i in range(3):
    print(random.random())
    print(random.randint(10, 20))
    print(random.choice(members))

class Dice():
    def roll(self):
        toss1 = random.randint(1, 6)
        toss2 = random.randint(1, 6)
        throw = (toss1, toss2)
        return throw # instead of defining a tuple like "throw" we can directly use return toss1, toss2 (Python automatically returns a tuple.

throw1 = Dice()
print(throw1.roll())
