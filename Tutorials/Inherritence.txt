class Mammal:
    def __init__(self, animal_name):
        self.animal_name = animal_name

    def walk(self, animal_name):
        print(f'{self.animal_name} is running')


class Dog(Mammal):
    def __init__(self, dog_name, intensity, animal_name):
        self.dog_name = dog_name
        self.intensity = intensity
        self.animal_name = animal_name

    def bark(self, intensity, dog_name):
        print(f'{self.dog_name} is barking {self.intensity}')


class Cat(Mammal):
    def __init__(self, animal_name):
        self.animal_name = animal_name


dog1 = Dog(dog_name="Toby", intensity="Loudly", animal_name="Toby")
dog1.bark(dog_name="Toby", intensity="Loudly")
dog1.walk(animal_name="Toby")

cat1 = Cat(animal_name="Mia")
cat1.walk(animal_name="Mia")
