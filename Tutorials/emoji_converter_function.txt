def emoji_converter(message):
    words = message.split(' ')
    emojis = {
        ":)": "Happy Face",
        ":(": "Sad Face"
    }
    output = ""
    for word in words:
        output += emojis.get(word, word) + ' '
    return output


message = "Such a good day :)"
converted_message = emoji_converter(message=message)
print(converted_message)
