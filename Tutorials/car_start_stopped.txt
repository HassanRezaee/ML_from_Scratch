started = False
while True:
    command = input('What is your command: ').lower()
    if command == 'start':
        if started:
            print('Car started already.')
        else:
            print('Car started, ready to go ... ')
            started = True
    elif command == 'stop':
        if not started:
            print('Car already stopped.')
        else:
            print('Car stopped, get out now...')
            started = False
    elif command == 'help':
        print('Just follow the instructions.')
    elif command == 'quit':
        break
    else:
        print('I dont unsderstand')
