    print(f'Hi {first_name} {last_name}!')
    print('Welcome')


print("start")
greet_user(last_name='Smith', first_name='John')

# here John is the argument supplied to parameter first_name in function greet_user


# *** positional argument: one that are read in order they're fed to the function, e.g., greet_user('John', 'Smith')
#     keyword argument: one that we specify the parameters they're assigned to when calling the function
#     e.g., greet_user(fist_name='John', last_name='Smith')
#     For numerical values, always use keyword arguments since they add to the readability of the code:
