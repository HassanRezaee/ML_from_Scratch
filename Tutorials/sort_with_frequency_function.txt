### program to remove duplicates in a list
numbers = [1, 2, 3, 2, 1, 2, 4, 4, 5, 3, 2, 1, 2]
numbers_unique = []

for number in numbers:
    if number not in numbers_unique:
        numbers_unique.append(number)
        numbers_unique.append(numbers.count(number))


print(numbers_unique)
